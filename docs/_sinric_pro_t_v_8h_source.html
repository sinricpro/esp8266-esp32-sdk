65 bool SinricProTV::handleRequest(const char* deviceId, const char* action, JsonObject &request_value, JsonObject &response_value) {
66  if (strcmp(deviceId, this->deviceId) != 0) return false;
67  if (SinricProDevice::handleRequest(deviceId, action, request_value, response_value)) return true;
68 
69  bool success = false;
70  String actionString = String(action);
71 
72  if (volumeCallback && actionString == "setVolume") {
73  int volume = request_value["volume"];
74  success = volumeCallback(String(deviceId), volume);
75  response_value["volume"] = volume;
76  return success;
77  }
78 
79  if (adjustVolumeCallback && actionString == "adjustVolume") {
80  int volume = request_value["volume"];
81  success = adjustVolumeCallback(String(deviceId), volume);
82  response_value["volume"] = volume;
83  return success;
84  }
85 
86  if (muteCallback && actionString == "setMute") {
87  bool mute = request_value["mute"];
88  success = muteCallback(String(deviceId), mute);
89  response_value["mute"] = mute;
90  return success;
91  }
92 
93  if (mediaControlCallback && actionString == "mediaControl") {
94  String mediaControl = request_value["control"];
95  success = mediaControlCallback(String(deviceId), mediaControl);
96  response_value["control"] = mediaControl;
97  return success;
98  }
99 
100  if (selectInputCallback && actionString == "selectInput") {
101  String input = request_value["input"];
102  success = selectInputCallback(String(deviceId), input);
103  response_value["input"] = input;
104  return success;
105  }
106 
107  if (actionString == "changeChannel") {
108 
109  if (changeChannelCallback && request_value["channel"].containsKey("name")) {
110  String channelName = request_value["channel"]["name"] | "";
111  success = changeChannelCallback(String(deviceId), channelName);
112  response_value["channel"]["name"] = channelName;
113  }
114 
115  if (changeChannelNumberCallback && request_value["channel"].containsKey("number")) {
116  int channelNumber = request_value["channel"]["number"];
117  String channelName("");
118  success = changeChannelNumberCallback(String(deviceId), channelNumber, channelName);
119  response_value["channel"]["name"] = channelName;
120  }
121  return success;
122  }
123 
124  if (skipChannelsCallback && actionString == "skipChannels") {
125  int channelCount = request_value["channelCount"] | 0;
126  String channelName;
127  success = skipChannelsCallback(String(deviceId), channelCount, channelName);
128  response_value["channel"]["name"] = channelName;
129  return success;
130  }
131 
132  return success;
133 }
